1650 {

1652  uint32_t PreemptPriorityBits;

1654 
1655  PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;

1657

1660  ((SubPriority & ((1 << (SubPriorityBits )) - 1)))
1661  );
