int ClkCfg(int iCd, int iClkSrc, int iSysClockDiv, int iClkOut) ==========Configures clock system.

Parameters
iCd:{CLK_CD0,CLK_CD1,CLK_CD2,CLK_CD3,CLK_CD4,CLK_CD5,CLK_CD6,CLK_CD7}
  • 0 or CLK_CD0 to divide clock by 1.
  • 1 or CLK_CD1 to divide clock by 2.
  • 2 or CLK_CD2 to divide clock by 4.
  • 3 or CLK_CD3 to divide clock by 8.
  • 4 or CLK_CD4 to divide clock by 16.
  • 5 or CLK_CD5 to divide clock by 32.
  • 6 or CLK_CD6 to divide clock by 64.
  • 7 or CLK_CD7 to divide clock by 128.
iClkSrc:{CLK_HF,CLK_LFX,CLK_LF,CLK_P4}
  • 0 or CLK_HF for 16MHz clock.
  • 1 or CLK_LFX for external 32kHz clock.
  • 2 or CLK_LF for internal 32kHz clock.
  • 3 or CLK_P4 for for clocking from pin P1.0.
iSysClockDiv:{CLKSYSDIV_DIV2EN_DIS,CLKSYSDIV_DIV2EN}
  • 0 or CLKSYSDIV_DIV2EN_DIS for 16MHz clock when internal HF oscillator is selected
  • 1 or CLKSYSDIV_DIV2EN_EN for 8MHz clock when internal HF oscillator is selected
iClkOut:{CLK_UCLKCG,CLK_UCLK,CLK_UDIV,CLK_HFO,CLK_LFO,CLK_LFXO}
  • 0 or CLK_UCLKCG to output UCLKCG clock.
  • 1 or CLK_UCLK to output UCLK clock.
  • 2 or CLK_UDIV to output UCLK divided by CD clock.
  • 5 or CLK_HFO to output 16MHz clock.
  • 6 or CLK_LFO to output internal 32kHz clock.
  • 7 or CLK_LFXO to output external 32kHz clock.
Returns
1
Examples:
ADC0_DMA.c, ADC1_DMA.c, ADCMeter.c, Blink.c, DAC_DMA.c, FlashProtect.c, FlashSign.c, FlashWrite.c, I2CDMA_Master.c, I2CDMA_Slave.c, I2Cloopback.c, I2Cmaster.c, I2CmasterDMA.c, I2Cslave.c, I2CslaveDMA.c, PowerDown.c, PWM.c, RTD_Demo.c, RTD_toSpi.c, SPIDMA_Master.c, SPIDMA_Slave.c, Thermocouple_to_DAC.c, Thermocouple_to_UART.c, Timers.c, UARTDma_Demo.c, and UartDmaTx.c.
int ClkDis ( int  iClkDis)

int ClkDis(int iClkDis) ==========Disables Clock to Peripheral blocks

Parameters
iClkDis:{0|CLKDIS_DISSPI0CLK| CLKDIS_DISSPI1CLK| CLKDIS_DISI2CCLK| CLKDIS_DISUARTCLK| CLKDIS_DISPWMCLK| CLKDIS_DIST0CLK| CLKDIS_DIST1CLK| CLKDIS_DISDACCLK| CLKDIS_DISDMACLK| CLKDIS_DISADCCLK}
  • CLKDIS
  • Combination of the following features :
    • 0 or Enable clock to all peripherals
    • 1 or DISSPI0CLK disable SPI0 peripheral Clock
    • 2 or DISSPI1CLK disable SPI0 peripheral Clock
    • 4 or DISI2CCLK disable I2C peripheral Clock
    • 8 or DISUARTCLK disable UART peripheral Clock
    • 16 or DISPWMCLK disable PWM peripheral Clock
    • 32 or DISSPI1CLK disable SPI0 peripheral Clock
    • 64 or DIST0CLK disable T0 peripheral Clock
    • 128 or DIST1CLK disable T1 peripheral Clock
    • 256 or DISDACCLK disable DAC peripheral Clock
    • 512 or DISDMACLK disable DMA peripheral Clock
    • 1024 or DISADCCLK disable ADC peripheral Clock
Returns
1.
Examples:
ADC0_DMA.c, ADC1_DMA.c, ADCMeter.c, Blink.c, DAC_DMA.c, FlashProtect.c, FlashSign.c, FlashWrite.c, I2CDMA_Master.c, I2CDMA_Slave.c, I2Cloopback.c, I2Cmaster.c, I2CmasterDMA.c, I2Cslave.c, I2CslaveDMA.c, PowerDown.c, PWM.c, RTD_Demo.c, RTD_toSpi.c, SPIDMA_Master.c, SPIDMA_Slave.c, Thermocouple_to_DAC.c, Thermocouple_to_UART.c, Timers.c, UARTDma_Demo.c, and UartDmaTx.c.
